  mng integer,
  worker integer,
  type integer,
  mng_grade integer,
  mng_memo varchar(255),
  worker_grade integer,
  worker_memo varchar(255),

);

  user integer,
  company varchar(255),
  job integer,
  job_other varchar(255),
  experience varchar(255),
  from timestamp,

);
